\begin{frame}{Capstone Project Prerequisites}
\textbf{What you need to know before starting}

\begin{columns}
\begin{column}{0.5\textwidth}
\textbf{Technical Prerequisites:}

\begin{block}{Programming (Required)}
\begin{itemize}
\item \textbf{Python:} Proficient (read/write/debug)
\item \textbf{Libraries:} pandas, numpy, scikit-learn
\item \textbf{Jupyter:} Notebooks and lab environments
\item \textbf{Git:} Version control basics (clone, commit, push)
\end{itemize}

\begin{block}{Mathematics (Required)}
\begin{itemize}
\item \textbf{Statistics:} Hypothesis testing, confidence intervals
\item \textbf{Probability:} Distributions, expectation, conditional probability
\item \textbf{Linear Algebra:} Matrix operations, eigenvalues
\item \textbf{Calculus:} Derivatives, gradients, optimization
\end{itemize}

\begin{itemize}
\item Supervised learning (regression, classification)

\begin{frame}{Getting Help: Resources and Strategies}
\textbf{How to get unstuck effectively}

\begin{columns}
\begin{column}{0.5\textwidth}
\textbf{The 30-Minute Rule:}

\begin{block}{When facing a problem...}
\begin{enumerate}
\item \textbf{0-15 min:} Try to solve it yourself
   \begin{itemize}
   \item Read error messages carefully
   \item Google the specific error
   \item Check documentation
   \item Review lecture notes
   \end{itemize}

\item \textbf{15-30 min:} Structured debugging
   \begin{itemize}
   \item Isolate the problem (minimal example)
   \item Print intermediate values
   \item Test components individually
   \item Check assumptions
   \end{itemize}

\item \textbf{After 30 min:} Ask for help!
   \begin{itemize}
   \item Don't waste hours being stuck
   \item Document what you've tried
   \item Formulate specific question
   \end{itemize}
\end{enumerate}
\end{block}

\vspace{0.2cm}

\textbf{Where to Ask:}
\begin{enumerate}
\item Course Slack: \#capstone-help
\item Office hours (in-person/virtual)
\item Peer study groups
\item Stack Overflow (for general questions)
\end{enumerate}
\end{column}

\begin{column}{0.5\textwidth}
\textbf{How to Ask Good Questions:}

\begin{block}{Bad Question}
\textcolor{crimson}{"My code doesn't work. Help!"}

\textbf{Problem:} Vague, no context, no effort shown
\end{block}

\begin{block}{Good Question}
\textcolor{forest}{"I'm getting a KeyError in my feature engineering function (line 42). I've verified the column exists in the DataFrame. I've tried [X, Y, Z]. Here's a minimal example: [code snippet]. Any ideas?"}

\textbf{Why good:} Specific error, shows effort, provides context
\end{block}

\vspace{0.2cm}

\textbf{Question Template:}

\begin{enumerate}
\item \textbf{Context:} What are you trying to do?
\item \textbf{Problem:} What's going wrong? (error message)
\item \textbf{Attempted:} What have you tried?
\item \textbf{Code:} Minimal reproducible example
\item \textbf{Question:} Specific question
\end{enumerate}

\vspace{0.2cm}

\textbf{Office Hours Etiquette:}
\begin{itemize}
\item Come prepared with specific questions
\item Share screen/code in advance if possible
\item Take notes during discussion
\item Follow up if solution works
\end{itemize}
\end{column}
\end{columns}
